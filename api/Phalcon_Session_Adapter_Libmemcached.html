Class Phalcon\Session\Adapter\Libmemcached¶

extends abstract class Phalcon\Session\Adapter

implements ArrayAccess, Traversable, IteratorAggregate, Countable, Phalcon\Session\AdapterInterface

Source on GitHub

This adapter store sessions in libmemcached

<?php

 $session = new Phalcon\Session\Adapter\Libmemcached(array(
     'servers' => array(
         array('host' => 'localhost', 'port' => 11211, 'weight' => 1),
     ),
     'client' => array(
         Memcached::OPT_HASH => Memcached::HASH_MD5,
         Memcached::OPT_PREFIX_KEY => 'prefix.',
     ),
    'lifetime' => 3600,
    'prefix' => 'my_'
 ));

 $session->start();

 $session->set('var', 'some-value');

 echo $session->get('var');
